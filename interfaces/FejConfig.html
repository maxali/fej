Interface FejConfig

Configuration for creating a Fej instance

Example

const config: FejConfig = {
  retry: {
    attempts: 3,
    delay: 1000,
    backoff: 'exponential',
  },
  errorTransforms: [
    async (error, ctx) => {
      console.error(`Request failed: ${ctx.request.url}`);
      return error;
    },
  ],
};
const api = createFej(config);
